// GÖRSELİ İNDİRME FONKSİYONU - GÜÇLENDİRİLMİŞ VERSİYON downBtn.onclick = () => { // Waveform içindeki tüm çizim katmanlarını bul const canvases = document.querySelectorAll('#waveform canvas'); if (canvases.length > 0) { // En üstteki ana çizimi al const mainCanvas = canvases[0]; try { // Resmi PNG formatına çevir const dataUrl = mainCanvas.toDataURL('image/png'); // İndirme işlemini tetiklemek için geçici bir 'a' etiketi oluştur const link = document.createElement('a'); link.href = dataUrl; // Dosya ismini tarihli yaparak karışıklığı önle const date = new Date().getTime(); link.download = `sonikskin-dalga-${date}.png`; // Sayfaya ekle, tıkla ve sil (Mobil uyumluluk için şart) document.body.appendChild(link); link.click(); document.body.removeChild(link); } catch (e) { console.error("Görsel oluşturma hatası:", e); alert("Görsel hazırlanırken bir sorun oluştu. Lütfen ekran görüntüsü almayı deneyin."); } } else { alert("Henüz bir ses dalgası oluşturulmadı!"); } };
