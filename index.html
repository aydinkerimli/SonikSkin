class SporYonetimi: def __init__(self): self.aktiviteler = [] def aktivite_kaydet(self, ad, dakika, kalori): self.aktiviteler.append({ "ad": ad, "sure": dakika, "kalori": kalori }) print(f"{ad} kaydedildi.") def rapor_olustur(self): print("\n--- SPOR ÖZETİ ---") toplam = 0 for a in self.aktiviteler: print(f"{a['ad']}: {a['sure']} dk - {a['kalori']} kcal") toplam += a['kalori'] print(f"Toplam: {toplam} kcal\n") # Çalıştırma takip = SporYonetimi() takip.aktivite_kaydet("Kardiyo", 40, 400) takip.aktivite_kaydet("Ağırlık", 50, 250) takip.rapor_olustur()
