class SporTakip: def __init__(self, kullanici_adi): self.kullanici = kullanici_adi self.egzersizler = [] def egzersiz_ekle(self, isim, sure, kalori): # Gereksiz tüm komutlar ve notlar silindi. yeni_kayit = { "egzersiz": isim, "sure_dk": sure, "yakilan_kalori": kalori } self.egzersizler.append(yeni_kayit) return f"{isim} başarıyla eklendi." def ozet_goster(self): print(f"--- {self.kullanici} İçin Spor Özeti ---") for e in self.egzersizler: print(f"- {e['egzersiz']}: {e['sure_dk']} dk, {e['yakilan_kalori']} kcal") # Kullanım: sporcu = SporTakip("Mert") sporcu.egzersiz_ekle("Koşu", 30, 300) sporcu.egzersiz_ekle("Yüzme", 45, 500) sporcu.ozet_goster()
